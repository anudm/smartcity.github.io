BUBBLE SORT
#include <iostream>
#include <vector>
using namespace std;

void bubbleSort(vector<int>& A) {
    int n = A.size();
    for (int i = 0; i < n - 1; ++i) {
        for (int j = 0; j < n - 1 - i; ++j) {
            if (A[j + 1] < A[j]) {
                swap(A[j], A[j + 1]);
            }
        }
    }
}

int main() {
    int n;
    cout << "Enter the number of elements: ";
    cin >> n;
    vector<int> arr(n);
    cout << "Enter the elements: ";
    for (int i = 0; i < n; ++i) 
    cin >> arr[i];

    bubbleSort(arr);

    cout << "Sorted Array (Bubble Sort): ";
    for (int x : arr) cout << x << " ";
    return 0;
}
